#include <iostream>
#include <algorithm>
using namespace std;
int d[5001];
int main() {
	cin.tie(0);
	ios_base::sync_with_stdio(0);

	int n;
	cin >> n;
	d[3] = d[5] = 1;
	for (int i = 6; i <= n; i++) {
		int a = d[i - 3];
		int b = d[i - 5];
		if (a == 0)
			a = 10000000;
		if (b == 0)
			b = 10000000;
		int m = min(a, b);
		if (m != 10000000)
			d[i] = m + 1;
		

	}
	
	if (d[n] != 0)
		cout << d[n];
	else
		cout << -1;

}
